        // MOCKED AI FUNCTION to replace live API call
        async function callGemini(prompt) {
            // Simulate network delay to keep the shimmer UI effect
            await new Promise(resolve => setTimeout(resolve, 1500));

            const lowerPrompt = prompt.toLowerCase();

            // Return hardcoded insights based on keywords
            if (lowerPrompt.includes("sector") || lowerPrompt.includes("45%")) {
                return "Based on the data, the 45% road transport share is critical because it represents the largest, most easily electrified segment of oil demand. As EV adoption follows an S-curve and battery costs drop below $80/kWh, the displacement of this specific 45% will accelerate rapidly, forcing global crude demand to peak around 2029 before heavy alternatives even scale.";
            } else if (lowerPrompt.includes("displacement") || lowerPrompt.includes("risk") || lowerPrompt.includes("trend")) {
                return "The risk for oil producers is severe. Currently, EVs displace 1.8 mb/d, which is manageable. However, as heavy hydrogen tech scales post-2035, combined displacement reaches 27 mb/d by 2050. Producers will face a structurally shrinking market, forcing refinery consolidation and leaving high-cost upstream assets stranded.";
            } else if (lowerPrompt.includes("net zero") || lowerPrompt.includes("residual") || lowerPrompt.includes("scenario")) {
                return "A 100% transition is unlikely because Net Zero does not mean 'zero oil.' The 'Residual 25%' (approx. 24 mb/d) consists of petrochemicals, asphalt, and specialized lubricants—materials essential for modern infrastructure that lack scalable non-fossil substitutes. Achieving Net Zero requires capturing these residual emissions, not eliminating the oil production entirely.";
            } else if (lowerPrompt.includes("2029") || lowerPrompt.includes("peak")) {
                return "The 2029 peak is driven by the convergence of battery cost parity and global fleet turnover. Once EVs hit mass-market affordability globally, the erosion of passenger gasoline demand will permanently outpace the growth of oil demand in emerging markets.";
            } else if (lowerPrompt.includes("hydrogen") || lowerPrompt.includes("h2")) {
                return "Hydrogen and its derivatives (like ammonia) are positioned to decarbonize 'hard-to-abate' sectors like heavy trucking and shipping where batteries are too heavy. This transition is expected to scale meaningfully in the mid-to-late 2030s once green hydrogen production achieves cost parity with diesel.";
            } else {
                return "According to the report, global oil demand is currently 103 mb/d and is expected to peak by 2029. While EVs and Hydrogen will significantly displace combustible fuels over time, a 'Residual 25%' floor of 24 mb/d will remain by 2050 for petrochemicals and industrial materials.";
            }
        }

        async function generateInsight(sectionId) {
            const modal = document.getElementById('ai-modal');
            const content = document.getElementById('modal-content');
            const title = document.getElementById('modal-title');
            
            modal.classList.remove('hidden');
            content.innerHTML = '<div class="ai-shimmer h-4 w-full mb-2 rounded"></div><div class="ai-shimmer h-4 w-3/4 mb-2 rounded"></div>';
            
            let prompt = "";
            if (sectionId === 'sector') {
                title.innerText = "✨ AI Sector Vulnerability Analysis";
                prompt = "Analyze the sector vulnerability chart. Why is the 45% road transport share the most critical factor for oil demand peak in 2029?";
            } else if (sectionId === 'displacement') {
                title.innerText = "✨ AI Trend Forecast";
                prompt = "Based on the displacement data (1.8mb/d now to 27mb/d by 2050), explain the risk for oil producers in the mid-2030s when heavy hydrogen tech begins to scale.";
            } else {
                title.innerText = "✨ AI Scenario Analysis";
                prompt = "Compare the Net Zero scenario with Business-as-Usual. Explain the 'Residual 25%' and why it makes a 100% transition unlikely.";
            }

            const insight = await callGemini(prompt);
            content.innerText = insight;
        }

        async function askAI() {
            const input = document.getElementById('ai-input');
            const history = document.getElementById('ai-chat-history');
            const query = input.value.trim();
            if (!query) return;

            const userDiv = document.createElement('div');
            userDiv.className = "bg-[#0f766e]/10 p-2 rounded-lg ml-4 text-right border-r-2 border-[#0f766e]";
            userDiv.innerText = query;
            history.appendChild(userDiv);
            input.value = "";
            history.scrollTop = history.scrollHeight;

            const loadingDiv = document.createElement('div');
            loadingDiv.className = "ai-shimmer h-10 w-2/3 rounded-lg p-2 mt-4";
            history.appendChild(loadingDiv);
            history.scrollTop = history.scrollHeight;

            const response = await callGemini(query);
            loadingDiv.remove();

            const aiDiv = document.createElement('div');
            aiDiv.className = "bg-white p-3 rounded-lg shadow-sm border border-stone-200 text-stone-800 mt-4";
            aiDiv.innerText = response;
            history.appendChild(aiDiv);
            history.scrollTop = history.scrollHeight;
        }

        // Chart definitions
        const wrapLabels = (labels) => {
            return labels.map(label => {
                if (label.length <= 16) return label;
                const words = label.split(' ');
                let lines = [];
                let currentLine = '';
                words.forEach(word => {
                    if ((currentLine + word).length > 16) {
                        if (currentLine) lines.push(currentLine.trim());
                        currentLine = word + ' ';
                    } else {
                        currentLine += word + ' ';
                    }
                });
                if (currentLine) lines.push(currentLine.trim());
                return lines;
            });
        };

                        return label.join(' ');
                    } else {
                        return label;
                    }
                }
            }
        };
